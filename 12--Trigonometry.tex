% COMMAND NAME CONVENTIONS

% - First letter represents typesetting mode: 
%       -- Math form (prefixed with 'm')
%       -- Inline form (prefixed with 'i')

% mtusin/itusin
% Math - Trig - Unity of Sine
%
%       sin^b a
%   1 = -------
%       sin^b a
%
%   arg a: angle
