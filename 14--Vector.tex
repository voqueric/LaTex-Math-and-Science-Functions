\newcommand{ \ivnteb }[1]{ \mvnteb{ #1 } }

% mvdotgv/ivdotgv
% Math - Vect - Vector Dot Product Geometric Definition using Vectors
%
%   ->  ->    ->    ->
%   a * b = | a | | b | cos c
%
%   arg a: vector a
%   arg b: vector b
%   arg c: angle between vectors
\newcommand{ \mvdotgv }[3]{ \vert \mvnteb{ #1 } \vert \vert \mvnteb{ #2 } \vert \cos{ #3 } }
\newcommand{ \ivdotgv }[3]{ \mvdotgv{ #1 }{ #2 }{ #3 } }


% mvdotgva/ivdotgva
% Math - Vect - Vector Dot Product Geometric Definition using Vectors Angle Solution
%
%                 ->  ->
%                 a * b
%      cos c = -----------
%                ->    ->
%              | a | | b |
%
%   arg a: vector a
%   arg b: vector b
\newcommand{ \mvdotgva }[3]{ \dfrac{ \mvnteb{ #1 } \cdot \mvnteb{ #2 } }{ \vert \mvnteb{ #1 } \vert \vert \mvnteb{ #2 } } }
\newcommand{ \ivdotgva }[3]{ \frac{ \ivnteb{ #1 } \cdot \ivnteb{ #2 } }{ \vert \ivnteb{ #1 } \vert \vert \ivnteb{ #2 } } }


% mvdotgm/ivdotgm
% Math - Vect - Vector Dot Product Geometric Definition using Vector Magnitudes
%
%   ->  ->
%   a * b = a b cos c
%
%   arg a: magnitude vector a
%   arg b: magnitude vector b
%   arg c: angle between vectors
\newcommand{ \mvdotgm }[3]{ { #1 }{ #2 } \cos{ #3 } }
\newcommand{ \ivdotgm }[3]{ \mvdotgm{ #1 }{ #2 }{ #3 } }


% mvcrsgv/ivcrsgv

%   arg a: magnitude vector a
%   arg b: magnitude vector b
%   arg c: angle between vectors
\newcommand{ \mvcrsgv }[3]{ \vert #1 \vert \vert #2 \vert \sin{ #3 } \mvhatb{ n } }
\newcommand{ \ivcrsgv }[3]{ \mvcrsgv{ #1 }{ #2 }{ #3 } }


% mvcrsgm/ivcrsgm

\newcommand{ \ivcrsgm }[3]{ \mvcrsgm{ #1 }{ #2 }{ #3 } }

% mvccrttbl/ivccrttbl
% Math - Vect - Curl Table
%   
%           | ^   ^   ^ |
%           | a   b   c |
%   ->  ->  |           |
%   l x r = | d   e   f |
%           |           |
%           | g   h   i |
%
%   arg a: unit vector #1
%   arg b: unit vector #2
%   arg c: unit vector #3
%
%   arg d: left vector component #1
%   arg e: left vector component #2
%   arg f: left vector component #3
%
%   arg g: right vector component #1
%   arg h: right vector component #2
%   arg i: right vector component #3
\newcommand{ \mvccrttbl }[9]{
    \begin{vmatrix}
        #1    & #2    & #3    \\
        #4    & #5    & #6    \\
        #7    & #8    & #9
    \end{vmatrix}
    }
\newcommand{ \ivccrttbl }[9]{ \mvccrttbl{ #1 }{ #2 }{ #3 }{ #4 }{ #5 }{ #6 }{ #7 }{ #8 }{ #9 } }


% mvccrtdet
% Math - Vect - Curl Determinants
%   
%   ->  ->  | e   f | ^   | d   f | ^   | d   e | ^
%   l x r = |       | a - |       | b + |       | c
%           | h   i |     | g   i |     | g   h |
%
%   arg a: unit vector #1
%   arg b: unit vector #2
%   arg c: unit vector #3
%
%   arg d: left vector component #1
%   arg e: left vector component #2
%   arg f: left vector component #3
%
%   arg g: right vector component #1
%   arg h: right vector component #2
%   arg i: right vector component #3
\newcommand{ \mvccrtdet }[9]{
    \begin{vmatrix}
        #5      & #6    \\
        #8      & #9
    \end{vmatrix} #1
    - 
    \begin{vmatrix}
        #4      & #6    \\
        #7      & #9
    \end{vmatrix} #2
    +
    \begin{vmatrix}
        #4      & #5    \\
        #7      & #8
    \end{vmatrix} #3
    }
    

% ivccrtdet
% Math - Vect - Curl Determinants
%   
%   ->  ->              ^               ^               ^  
%   l x r = (e i - h f) a - (d i - g f) b + (d h - g e) c
%
%   arg a: unit vector #1
%   arg b: unit vector #2
%   arg c: unit vector #3
%
%   arg d: left vector component #1
%   arg e: left vector component #2
%   arg f: left vector component #3
%
%   arg g: right vector component #1
%   arg h: right vector component #2
%   arg i: right vector component #3
\newcommand{ \ivccrtdet }[9]{ (#5 #9 - #8 #6){ #1 } - (#4 #9 - #7 #6){ #2 } + (#4 #8 - #7 #5){ #3 } }


% mvlcrt/ivlcrt
% Math - Vect - Cartesian Laplacian
%
%            d^2       d^2       d^2

%           d^2 a     d^2 b     d^2 c
%
%   arg a: operator variable #1
%   arg b: operator variable #2
%   arg c: operator variable #3
%   arg d: function
\newcommand{ \mvlcrt }[4]{ \mcpdelb{ #1 }{ 2 }{ #4 } + \mcpdelb{ #2 }{ 2 }{ #4 } + \mcpdelb{ #3 }{ 2 }{ #4 } }
\newcommand{ \ivlcrt }[4]{ \icpdelb{ #1 }{ 2 }{ #4 } + \icpdelb{ #2 }{ 2 }{ #4 } + \icpdelb{ #3 }{ 2 }{ #4 } }

\newcommand{ \ivgcyllng }[3]{ \,\frac{ 1 }{ #2 } \ivgcrtoneg{ #1 }{ #1 }{ #3 } }

\newcommand{ \ivgcylhgt }[2]{ \ivgcrtoneb{ #1 }{ #1 }{ #2 } }


% mvgcyl/ivgcyl
% Math - Vect - Cylindrical Gradient Composite
%
%    d    ^   1  d    ^    d    ^
%   --- d a + - --- d b + --- d c
%   d a       b d b       d c
%

\newcommand{ \ivgcyl }[4]{ ( \ivgcylrad{ #1 }{ #4 } + \ivgcyllng{ #2 }{ #1 }{ #4 } + \ivgcylhgt{ #3 }{ #4 } ) }


% mvdifdcyl/ivdifdcyl

\newcommand{ \ivdifdcyl }[3]{ \mvdifcyldis{ #1 }{ #2 }{ #3 } }

% mvtsrthrtbl/ivtsrthrtbl

%       |           |

%       |           |
%       | g   h   i |
%
%   arg a: tensor r1, c1
%   arg b: tensor r1, c2
%   arg c: tensor r1, c3
%
%   arg d: tensor r2, c1

    \end{bmatrix}
    }
\newcommand{ \ivtsrthrtbl }[9]{ \mvtsrthrtbl{ #1 }{ #2 }{ #3 }{ #4 }{ #5 }{ #6 }{ #7 }{ #8 }{ #9 } }
