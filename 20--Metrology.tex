% mdsnte/idsnte
% Metrology - Dimensioning - Scientific Notation
%
%   a x 10^b
%
%   arg a:   decimal
%   arg b:   exponent
\newcommand{ \mmdsnte }[2]{ #1 \times 10^{ #2 } }
\newcommand{ \imdsnte }[2]{ \mmdsnte{ #1 }{ #2 } }

% mdsntepmt/idsntepmt
% Metrology - Dimensioning - Scientific Notation with +/- Tolerance
%
%   (a +/- b) x 10^c
%
%   arg a: decimal
%   arg b: tolerance 
%   arg c: exponent
\newcommand{ \mdsntepmt }[3]{ ( #1 \pm #2 ) \times 10^{ #3 } }
\newcommand{ \idsntepmt }[3]{ \mdsntepmt{ #1 }{ #2 }{ #3 } }


% mdadis/idadis
% Metrology - Dimensioning - Analysis Displacement

% mdastr/idastr
% Metrology - Dimensioning - Analysis Strecth
%
%         L
%   s {=} -

%

\newcommand{ \idastr }[0]{ \mdadis{ } \mdadis{ }^{ -1 } }


% mdavel/idavel
% Metrology - Dimensioning - Analysis Velcocity
%
%         L

%         T
%
%     {=} L T^-1
%

\newcommand{ \idavel }[0]{ \mdadis{ } \mdatime{ }^{ -1 }  }

\newcommand{ \idaacc }[0]{ \mdadis{ } \mdatime{ }^{ -2 }  }

\newcommand{ \idavis }[0]{ \mdamass{ } \mdadis{ }^{ -1 } \mdatime{ }^{ -1 } }
